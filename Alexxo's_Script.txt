end

local function randomWait(base, variance)
    if not ANTI_BAN then return task.wait(base) end
    local factor = 1 + (math.random() * 2 - 1) * (variance or 0.2)
    task.wait(math.max(0.05, base * factor))
end

local function jitter(amount)
    if not ANTI_BAN then return Vector3.new(0,0,0) end
    return Vector3.new((math.random()*2-1)*amount, (math.random()*2-1)*amount, (math.random()*2-1)*amount)
end

local function censorName(name)
    if not name or name == "" then return "Unknown" end
    if #name <= 2 then return name end
    return string.sub(name, 1, 1) .. string.rep("*", #name - 2) .. string.sub(name, -1)
end

local function sendWebhook(message, eventType, extraData)
    if WebhookUrlBox then
        webhookSettings.url = WebhookUrlBox.Text
    end
    if not webhookSettings.enabled then return false, "Webhook disabled" end
    local url = webhookSettings.url:gsub("%s+", "")
    if url == "" then return false, "URL empty" end
    if not string.find(url, "discord.com/api/webhooks/") then return false, "Invalid Discord URL" end

    local playerName = censorName(Player.Name)
    local displayName = censorName(Player.DisplayName)
    local currentTime = os.date("%Y-%m-%d %H:%M:%S")
    
    local embed = {
        title = "Asura Farm Notification",
        color = eventType == "death" and 0xFF0000 or (eventType == "kyubite" and 0x00FF00 or 0x3498DB),
        fields = {},
        footer = {text = "Asura Farm Monitor • " .. os.date("%x")},
        timestamp = DateTime.now():ToIsoDate()
    }

    if eventType == "death" then
        embed.fields = {
            {name = "Event", value = "💀 You died!", inline = false},
            {name = "Session Time", value = SessionTimeLabel.Text, inline = true},
            {name = "Bosses Killed", value = tostring(bossesDefeated), inline = true},
            {name = "Money", value = MoneyLabel.Text, inline = true},
            {name = "Kyubite", value = KyubiteLabel.Text, inline = true},
            {name = "Player", value = string.format("%s (@%s)", displayName, playerName), inline = true}
        }
    elseif eventType == "kyubite" then
        embed.fields = {
            {name = "Event", value = "✨ Kyubite Gained", inline = false},
            {name = "New Total", value = tostring(kyubiteData.current), inline = true},
            {name = "Obtained", value = "+" .. tostring(extraData.gain), inline = true},
            {name = "Session Time", value = SessionTimeLabel.Text, inline = true},
            {name = "Player", value = string.format("%s (@%s)", displayName, playerName), inline = true}
        }
    elseif eventType == "test" then
        embed.fields = {
            {name = "Test", value = "✅ Webhook is working properly!", inline = false},
            {name = "Player", value = string.format("%s (@%s)", displayName, playerName), inline = true},
            {name = "Time", value = currentTime, inline = true}
        }
    end

    local payload = {embeds = {embed}}
    local json = HttpService:JSONEncode(payload)
    local headers = {["Content-Type"] = "application/json"}

    local success, resp = pcall(function()
        return HttpService:RequestAsync({
            Url = url,
            Method = "POST",
            Headers = headers,
            Body = json
        })
    end)

    if success and resp.Success then
        return true, "Sent via HttpService"
    elseif success then
        return false, "HTTP " .. tostring(resp.StatusCode) .. ": " .. (resp.StatusMessage or "Unknown")
    end

    local syn = syn or _G.syn
    if syn and syn.request then
        success, resp = pcall(function()
            return syn.request({
                Url = url,
                Method = "POST",
                Headers = headers,
                Body = json
            })
        end)
        if success and resp.Success then
            return true, "Sent via syn.request"
        elseif success then
            return false, "HTTP " .. tostring(resp.StatusCode) .. ": " .. (resp.StatusMessage or "Unknown")
        else
            return false, "syn.request error: " .. tostring(resp)
        end
    end

    return false, "No webhook method available on this executor"
end

    local function selectTab(tab)
        if tab == "stats" then
            StatsTab.Visible = true
            SettingsTab.Visible = false
            WebhookTab.Visible = false
            TweenService:Create(statsTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(55,80,140)}):Play()
            TweenService:Create(settingsTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(45,45,55), TextColor3 = Color3.fromRGB(220,220,220)}):Play()
            TweenService:Create(webhookTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(45,45,55), TextColor3 = Color3.fromRGB(220,220,220)}):Play()
        elseif tab == "settings" then
            StatsTab.Visible = false
            SettingsTab.Visible = true
            WebhookTab.Visible = false
            TweenService:Create(statsTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(45,45,55), TextColor3 = Color3.fromRGB(220,220,220)}):Play()
            TweenService:Create(settingsTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(55,80,140), TextColor3 = Color3.fromRGB(255,255,255)}):Play()
            TweenService:Create(webhookTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(45,45,55), TextColor3 = Color3.fromRGB(220,220,220)}):Play()
        else
            StatsTab.Visible = false
            SettingsTab.Visible = false
            WebhookTab.Visible = true
            TweenService:Create(statsTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(45,45,55), TextColor3 = Color3.fromRGB(220,220,220)}):Play()
            TweenService:Create(settingsTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(45,45,55), TextColor3 = Color3.fromRGB(220,220,220)}):Play()
            TweenService:Create(webhookTabBtn, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(55,80,140), TextColor3 = Color3.fromRGB(255,255,255)}):Play()
        end
    end

local lastHopTime = 0

local function tryHttpRequest(url)
    local success, result = pcall(function() return HttpService:GetAsync(url) end)
    if success then return success, result end
    success, result = pcall(function()
        local req = HttpService:RequestAsync({Url = url, Method = "GET"})
        return req.Body
    end)
    if success then return success, result end
    success, result = pcall(function()
        local syn = syn or _G.syn
        if syn and syn.request then
            local resp = syn.request({Url = url, Method = "GET"})
            return resp.Body
        end
        error("syn.request not available")
    end)
    if success then return success, result end
    success, result = pcall(function()
        local http = http or _G.http
        if http and http.request then
            local resp = http.request(url)
            return resp
        end
        error("http.request not available")
    end)
    if success then return success, result end
    return false, "All HTTP methods failed"
end

local function rejoinWorld4()
    local maxAttempts = 15
    local attempt = 1
    local success = false

    while attempt <= maxAttempts and not success do
        HopStatus.Text = "Looking for World 4 ("..attempt.."/"..maxAttempts..")"
        local worldButton = nil
        for _, obj in ipairs(PlayerGui:GetDescendants()) do
            if obj:IsA("TextButton") then
                local text = obj.Text or ""
                if string.find(text, "IV") or string.find(text, "IV - Crimson Sakura Isles") or string.find(text, "Crimson Sakura Isles") then
                    worldButton = obj
                    break
                end
            end
        end

        if worldButton then
            currentAction = "Clicking World 4..."
            worldButton:Click()
            task.wait(2)
            if isInWorld4() then
                success = true
                HopStatus.Text = "Now in World 4"
                currentAction = "In World 4"
                break
            else
                HopStatus.Text = "World 4 not entered, retrying..."
                task.wait(1)
            end
        else
            HopStatus.Text = "World 4 button not found, waiting..."
            task.wait(2)
        end
        attempt = attempt + 1
    end

    if not success then
        HopStatus.Text = "⚠️ Re‑enter World 4 manually"
        currentAction = "Manual World 4 required"
        isRunning = false
    end
end

local function hopToLowServer()
    local now = tick()
    if now - lastHopTime < HOP_COOLDOWN then
        HopStatus.Text = "Cooldown " .. math.ceil(HOP_COOLDOWN - (now - lastHopTime)) .. "s"
        return
    end

    local maxPlayers = tonumber(MaxPlayersBox.Text) or 5
    if maxPlayers < 1 then maxPlayers = 1 end

    if isRunning then
        isRunning = false
        currentAction = "Paused for hop"
    end

    HopStatus.Text = "Fetching servers..."
    HopButton.Text = "Please wait..."
    HopButton.Active = false

    task.spawn(function()
        local success, result = pcall(function()
            local placeId = game.PlaceId
            local url = "https://games.roblox.com/v1/games/" .. placeId .. "/servers/Public?limit=100"
            local response = tryHttpRequest(url)
            if not response then return nil, "HTTP request failed" end
            local data = HttpService:JSONDecode(response)
            local servers = data.data
            if not servers or #servers == 0 then return nil, "No servers found" end

            local validServers = {}
            for _, server in ipairs(servers) do
                if server.playing < server.maxPlayers and server.playing >= 0 and server.playing <= maxPlayers then
                    table.insert(validServers, server)
                end
            end
            if #validServers == 0 then return nil, "No servers with ≤ "..maxPlayers.." players" end

            table.sort(validServers, function(a,b) return a.playing < b.playing end)

            local currentJobId = game.JobId
            for _, server in ipairs(validServers) do
                if server.id ~= currentJobId then
                    return server.id
                end
            end
            return validServers[1].id
        end)

        HopButton.Text = "LOW HOP SERVER"
        HopButton.Active = true

        if not success or not result then
            HopStatus.Text = "No low server, trying random..."
            lastHopTime = tick()
            pcall(function() TeleportService:Teleport(game.PlaceId, Player) end)
            task.spawn(rejoinWorld4)
            return
        end

        local jobId = result
        HopStatus.Text = "Teleporting to " .. jobId:sub(1,8) .. "..."
        lastHopTime = tick()
        pcall(function() TeleportService:TeleportToPlaceInstance(game.PlaceId, jobId, Player) end)
        task.spawn(rejoinWorld4)
    end)
end

local function findMoneyAsync()
    task.spawn(function()

local function safeTween(targetCFrame, speed, speedType)
    if not HumanoidRootPart then return end
    if ANTI_BAN then targetCFrame = targetCFrame + jitter(1.5) end
    local start = HumanoidRootPart.Position
    local target = targetCFrame.Position
    if target.Y < SAFE_Y then
        target = Vector3.new(target.X, SAFE_Y+SAFE_HEIGHT, target.Z)
        targetCFrame = CFrame.new(target, targetCFrame.LookVector)
    end
    local dist = (start - target).Magnitude
    local duration = math.max(dist / speed, 1.0)
    duration = math.clamp(duration, 1.0, 15)
    if start.Y < SAFE_Y then
        HumanoidRootPart.CFrame = CFrame.new(Vector3.new(start.X, SAFE_Y+20, start.Z))
        task.wait(0.1)
    end
    local tween = TweenService:Create(HumanoidRootPart, TweenInfo.new(duration, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {CFrame = targetCFrame})
    activeTween = tween
    activeTweenType = speedType
    activeTweenCF = targetCFrame
    local conn
    conn = RunService.Heartbeat:Connect(function()
        if not HumanoidRootPart then conn:Disconnect() return end
        if HumanoidRootPart.Position.Y < VOID_Y then
            HumanoidRootPart.Velocity = Vector3.new(0,0,0)
            HumanoidRootPart.CFrame = CFrame.new(Vector3.new(HumanoidRootPart.Position.X, SAFE_Y+30, HumanoidRootPart.Position.Z))
        end
    end)
    tween:Play()
    tween.Completed:Wait()
    conn:Disconnect()
    activeTween = nil
    activeTweenType = nil
    activeTweenCF = nil
end

local function safeTweenStepwise(targetCFrame, speed, stepDistance, speedType)
    stepDistance = stepDistance or STEP_DISTANCE
    if not HumanoidRootPart then return end
    local startPos = HumanoidRootPart.Position
    local targetPos = targetCFrame.Position
    local totalDist = (targetPos - startPos).Magnitude
    if totalDist <= stepDistance then 
        safeTween(targetCFrame, speed, speedType)
        return 
    end
    local direction = (targetPos - startPos).Unit
    local steps = math.ceil(totalDist / stepDistance)
    for i = 1, steps do
        if not HumanoidRootPart then break end
        local fraction = i / steps
        local waypointPos = startPos:Lerp(targetPos, fraction)
        waypointPos = Vector3.new(waypointPos.X, math.max(waypointPos.Y, SAFE_Y)+SAFE_HEIGHT, waypointPos.Z)
        safeTween(CFrame.new(waypointPos), speed, speedType)
        randomWait(0.15, 0.3)
    end
    local finalPos = Vector3.new(targetPos.X, math.max(targetPos.Y, SAFE_Y)+SAFE_HEIGHT, targetPos.Z)
    safeTween(CFrame.new(finalPos, targetCFrame.LookVector), speed, speedType)
end

local function huntBoss(bossHRP)
    if not HumanoidRootPart or not bossHRP then return end
    local offset = BEHIND_OFFSET
    if ANTI_BAN then offset = offset + jitter(1.0) end
    local target = bossHRP.Position + offset
    safeTweenStepwise(CFrame.new(target, bossHRP.Position), HUNT_SPEED, STEP_DISTANCE, "hunt")
end

local function moveToSpawn()
    if not HumanoidRootPart then return end
    local start = HumanoidRootPart.Position
    local target = TARGET_CFRAME.Position
    local dist = (start - target).Magnitude
    if dist > 800 then
        local steps = math.ceil(dist / 400)
        local dir = (target - start).Unit
        for i = 1, steps do
            if not HumanoidRootPart then return end
            local wp = start + (dir * 400 * i)
            wp = Vector3.new(wp.X, math.max(wp.Y, SAFE_Y)+SAFE_HEIGHT, wp.Z)
            safeTween(CFrame.new(wp), SPAWN_TRAVEL_SPEED, "spawn")
            randomWait(0.25, 0.3)
        end
    end
    local final = Vector3.new(target.X, math.max(target.Y, SAFE_Y)+SAFE_HEIGHT, target.Z)
    safeTween(CFrame.new(final), SPAWN_TRAVEL_SPEED, "spawn")    
end

        local cam = Workspace.CurrentCamera
        if cam and Humanoid then cam.CameraSubject = Humanoid; cam.CameraType = Enum.CameraType.Custom end
    end)
end

local function cleanupFight()
    if followConnection then followConnection:Disconnect() followConnection = nil end
    if bodyPos then pcall(function() bodyPos:Destroy() end) bodyPos = nil end
    if bodyGyro then pcall(function() bodyGyro:Destroy() end) bodyGyro = nil end
    if Humanoid then pcall(function() Humanoid.AutoRotate = true; Humanoid.PlatformStand = false end) end
    if HumanoidRootPart then pcall(function() HumanoidRootPart.Anchored = false end) end
    if activeTween then activeTween:Cancel() activeTween = nil end
    currentBoss = nil
    updateActivity()
end

local function isDead()
    if not Humanoid then return true end
    if Humanoid.Health <= 0 then return true end
    if not HumanoidRootPart or not HumanoidRootPart.Parent then return true end
    return false
end

local function waitForCharacter()
    while not HumanoidRootPart or not Humanoid or Humanoid.Health <= 0 do
        task.wait(0.5)
        if not isRunning then return false end
    end
    return true
end

            if tick()-lastActivityTime > 600 then
                VirtualUser:CaptureController()
                VirtualUser:ClickButton1(Vector2.new(0,0))
                lastActivityTime = tick()
            end
            task.wait(5)
        end
    end)
end

local function isBossOnFloor(bossHRP)
    if not bossHRP or not bossHRP.Parent then return false end
    local rp = RaycastParams.new()
    rp.FilterType = Enum.RaycastFilterType.Blacklist
    rp.FilterDescendantsInstances = {bossHRP.Parent, Character}
    rp.IgnoreWater = true
    local r = Workspace:Raycast(bossHRP.Position, Vector3.new(0,-50,0), rp)
    if r then return (bossHRP.Position - r.Position).Magnitude <= 30 end
    return false
end

local function getClosestBoss()
    local living = Workspace:FindFirstChild("Living")
    if not living then return nil end
    local closest, shortest = nil, math.huge
    for _,m in ipairs(living:GetChildren()) do
        if m:IsA("Model") and (m.Name:find("Asura") or m.Name:find("Incarnate")) then
            local hrp = m:FindFirstChild("HumanoidRootPart")
            local hum = m:FindFirstChild("Humanoid")
            if hrp and hum and hum.Health > 0 and hrp.Position.Y > 10 and isBossOnFloor(hrp) then
                local d = (HumanoidRootPart.Position - hrp.Position).Magnitude
                if d < shortest then shortest = d; closest = m end
            end
        end
    end
    return closest
end

local function checkForAdmins()
    for _, p in ipairs(Players:GetPlayers()) do
        for _, id in ipairs(ADMIN_IDS) do
            if p.UserId == id then
                currentAction = "Admin detected, leaving..."
                task.wait(1)
                hopToLowServer()
                return
            end
        end
    end
end

local function startBot()
    if isRunning then return end
    isRunning = true
    diedRecently = false
    sessionStartTime = tick()
    if StatusLabel and StatusLabel.Parent then
        StatusLabel.Text = "[Fighting]"
        StatusLabel.TextColor3 = Color3.fromRGB(0,255,0)
    end

    if deathConnection then deathConnection:Disconnect() end
    if Humanoid then
        deathConnection = Humanoid.Died:Connect(function()
            diedRecently = false
            if webhookSettings.enabled and webhookSettings.notifyDeath and webhookSettings.url ~= "" then
                task.spawn(function() sendWebhook("You died!", "death", nil) end)
            end
        end)
    end

    task.spawn(function()
        while isRunning do
            if diedRecently then
                currentAction = "You died! Click START to resume."
                cleanupFight()
                isRunning = false
                break
            end

            if not waitForCharacter() then break end

            local ok, err = pcall(function()
                if not servicesLoaded then randomWait(1,0.3) return end
                if isDead() then cleanupFight() randomWait(2,0.5) return end

                local boss = getClosestBoss()
                if not boss then
                    if currentBoss then
                        currentAction = "Returning to spawn"
                        cleanupFight()
                        if isDead() then randomWait(3,0.5) return end
                        returnToSpawn()
                        randomWait(0.5,0.3)
                    else
                        if isDead() then return end
                        currentAction = "Moving to spawn"
                        HumanoidRootPart.Anchored = false
                        moveToSpawn()
                        if isDead() then return end

                        HumanoidRootPart.Anchored = true
                        currentAction = "Spawning boss"
                        updateActivity()

                        local lock = RunService.RenderStepped:Connect(function()
                            if HumanoidRootPart then HumanoidRootPart.CFrame = TARGET_CFRAME end
                        end)

                        for i = 1, EXECUTION_COUNT1 do
                            if isDead() or not isRunning then break end
                            task.spawn(function()
                                if ANTI_BAN then task.wait(math.random()*0.1) end
                                pcall(function()
                                    ActivatePartyRF:InvokeServer()
                                    CreatePartyRF:InvokeServer("Asura's Incarnate")
                                    ProximityRF:InvokeServer(Workspace.Proximity.CreateParty)
                                end)
                            end)
                            task.wait(0.005)
                            updateActivity()
                        end

                        lock:Disconnect()
                        if HumanoidRootPart then HumanoidRootPart.Anchored = false end
                        randomWait(2.5,0.3)
                    end
                    return
                end

                currentBoss = boss
                local bossHRP = boss:WaitForChild("HumanoidRootPart")
                local bossHum = boss:WaitForChild("Humanoid")
                updateActivity()

                Humanoid.PlatformStand = false
                Humanoid.AutoRotate = false

                currentAction = "Approaching boss"
                huntBoss(bossHRP)
                if not isRunning then return end
                updateActivity()

                bodyPos = Instance.new("BodyPosition")
                bodyPos.MaxForce = Vector3.new(8000,8000,8000)
                bodyPos.D = 800
                bodyPos.P = 5000
                bodyPos.Parent = HumanoidRootPart

                bodyGyro = Instance.new("BodyGyro")
                bodyGyro.MaxTorque = Vector3.new(math.huge,math.huge,math.huge)
                bodyGyro.P = 3000
                bodyGyro.Parent = HumanoidRootPart

                local lastAttack = 0
                local lastHealth = bossHum.Health
                local lastPos = bossHRP.Position
                local transforming = false
                local stablePos = nil

                followConnection = RunService.Heartbeat:Connect(function(dt)
                    if not isRunning then return end
                    if isDead() then return end
                    if HumanoidRootPart.Position.Y < VOID_Y then
                        currentAction = "Fell, recovering"
                        cleanupFight()
                        pcall(function() safeTween(TARGET_CFRAME + Vector3.new(0,SAFE_HEIGHT,0), RETURN_SPEED, "return") end)
                        return
                    end
                    updateActivity()
                    if not boss.Parent or not bossHum or bossHum.Health <= 0 then return end
                    if not bossHRP or not bossHRP.Parent then return end
                    if not isBossOnFloor(bossHRP) then return end

                    local dist = (HumanoidRootPart.Position - bossHRP.Position).Magnitude
                    if dist > MAX_BOSS_DISTANCE then
                        currentAction = "Lost boss, returning"
                        cleanupFight()
                        local safe = bossHRP.Position + BEHIND_OFFSET + Vector3.new(0,SAFE_HEIGHT,0)
                        safeTween(CFrame.new(safe, bossHRP.Position), RETURN_SPEED, "return")
                        return
                    end

                    local curPos = bossHRP.Position
                    local curHealth = bossHum.Health
                    local moved = (curPos - lastPos).Magnitude

                    if moved < 0.03 and math.abs(curHealth - lastHealth) < 0.5 then
                        if not transforming then transforming = true; stablePos = HumanoidRootPart.Position end
                    else
                        if transforming then transforming = false; stablePos = nil end
                    end
                    lastPos = curPos
                    lastHealth = curHealth

                    if transforming then
                        currentAction = "Boss stuck, holding position"
                        local desired = bossHRP.Position + BEHIND_OFFSET
                        if stablePos then stablePos = stablePos:Lerp(desired, dt*3) else stablePos = desired end
                        bodyPos.Position = stablePos
                        bodyGyro.CFrame = CFrame.new(HumanoidRootPart.Position, bossHRP.Position)
                        local now = tick()
                        local attackDelay = ANTI_BAN and (ATTACK_DELAY * (0.8+math.random()*0.4)) or ATTACK_DELAY
                        if now - lastAttack >= attackDelay then
                            lastAttack = now
                            pcall(function() ToolActivatedRF:InvokeServer("Weapon",nil) end)
                        end
                    else
                        currentAction = "Fighting boss"
                        bodyPos.Position = bossHRP.Position + BEHIND_OFFSET
                        bodyPos.D = 1000
                        bodyGyro.CFrame = CFrame.new(HumanoidRootPart.Position, bossHRP.Position)

                        local now = tick()
                        local attackDelay = ANTI_BAN and (ATTACK_DELAY * (0.8+math.random()*0.4)) or ATTACK_DELAY
                        if now - lastAttack >= attackDelay then
                            lastAttack = now
                            pcall(function() ToolActivatedRF:InvokeServer("Weapon",nil) end)
                        end
                    end
                end)

                local checkStart = tick()
                while boss.Parent and bossHum and bossHum.Health > 0 and isRunning do
                    randomWait(0.05,0.5)
                    updateActivity()
                    if isDead() then break end
                    if tick()-checkStart > 180 then break end
                end

                if isRunning and boss.Parent and bossHum and bossHum.Health <= 0 then
                    bossesDefeated = bossesDefeated + 1
                end
                currentAction = "Boss defeated"
                currentBoss = nil
                cleanupFight()
                randomWait(0.5,0.3)
            end)

            if not ok then
                warn("Error: "..tostring(err))
                cleanupFight()
                randomWait(2,0.5)
            end
            randomWait(0.1,0.5)
        end
    end)
end

local function initChar()
    Character = Player.Character or Player.CharacterAdded:Wait()
    local t = tick()
    repeat
        HumanoidRootPart = Character:FindFirstChild("HumanoidRootPart")
        Humanoid = Character:FindFirstChild("Humanoid")
        task.wait(0.05)
    until (HumanoidRootPart and Humanoid) or (tick()-t > 5)
    if not HumanoidRootPart or not Humanoid then return false end
    setupCamera()
    return true
end
